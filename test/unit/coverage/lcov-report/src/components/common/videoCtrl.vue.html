    if (this.$route.name === 'courseVideo') {
      const videoPlayButton = document.querySelectorAll('.vjs-big-play-button')[0]
      videoPlayButton.addEventListener('touchend', () => {
        let fileId = this.$store.state.oces.videoTreeFirst
        if (fileId) {
          this.$store.dispatch('getVideoUrl', fileId).then(() => {
            if (this.$store.state.oces.videoUrl.Data.payload) {
              this.play(this.$store.state.oces.videoUrl.Data.payload.url)
            } else {
//          MessageBox('提示', this.$store.state.oces.videoUrl.Data.message)
            }
          })
        }
      }, false)
    }
  },
  components: {
    videoPlayer
  },

    player () {
      return this.$refs.videoPlayer.player
    }
  },

      if (src) {
        this.$refs.videoPlayer.player.src(src)
      }
      this.$refs.videoPlayer.player.play()

    },
    onPlayerFullScreenChange (player) {
    },
    // player is ready
    playerReadied (player) {
      // 解决fastclick 导致视频操作按钮延时问题
      for (let i = 0; i < document.querySelectorAll('.vjs-button').length; i++) {
        let ob = document.querySelectorAll('.vjs-button')[i]
        if (!ob.className.match(new RegExp('(\\s|^)' + 'needsclick' + '(\\s|$)'))) {
          ob.className = ob.className + ' ' + 'needsclick'
        }
      }
 
      player.on('fullscreenchange', function () {
        if (player.isFullscreen()) {
          window.screen.orientation.unlock()
        } else {
          window.screen.orientation.lock('portrait')
        }
      })
