  <div id="oces">
    <transition :name="transitionName">
      <router-view v-on:hideFooter="footHidden"></router-view>
    </transition>
    <footerCtrl v-show="showFoot"></footerCtrl>
  </div>
</template>
 
<script>
import footerCtrl from '@/components/footer/footerCtrl'
export default {
  name: 'oces',
  data () {
    return {
      transitionName: 'vux-pop-in',
      showFoot: true,
      isIOSGT7: false
    }
  },
  mounted () {
//     const u = navigator.userAgent
// //    const isAndroid = u.indexOf('Android') > -1 || u.indexOf('Adr') > -1 // android终端
//     const isiOS = !!u.match(/\(i[^;]+;( U;)? CPU.+Mac OS X/) // ios终端

//     ver = parseInt(ver[1], 10)
//     if (ver > 7 && isiOS) {
//       this.isIOSGT7 = true

  },
  watch: {
    '$route' (to, from) {
      const toDepth = to.path.split('/').length
      const fromDepth = from.path.split('/').length
      let transitionstr = 'vux-pop-in'
      if (from.name === 'home') {
        if (to.name === 'courses') {
          transitionstr = 'vux-pop-in'
        } else if (to.name === 'my') {
          transitionstr = 'vux-pop-in'
        }
      } else if (from.name === 'my') {
        if (to.name === 'mysetting') {
          transitionstr = 'vux-pop-in'
        } else if (to.name === 'home') {
          transitionstr = 'vux-pop-out'
        } else if (to.name === 'courses') {
          transitionstr = 'vux-pop-out'
        } else {
          transitionstr = toDepth < fromDepth ? 'vux-pop-out' : 'vux-pop-in'
        }
      } else if (from.name === 'courses') {
        if (to.name === 'home') {
          transitionstr = 'vux-pop-out'
        } else {
          transitionstr = 'vux-pop-in'
        }
      } else {
        transitionstr = toDepth < fromDepth ? 'vux-pop-out' : 'vux-pop-in'
      }
      this.transitionName = transitionstr
    }
  },
  components: {
    footerCtrl
  },
  methods: {
    footHidden (data) {
      this.showFoot = data
    }
  }
}
</script>
